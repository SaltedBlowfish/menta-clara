  <action>
1. Create `eslint.config.js` using ESLint 9 flat config format. MUST use ESM imports. The config enforces:
   - No `any` (typescript-eslint strict)
   - No unsafe assignments, calls, member access, returns
   - No `as` type assertions (`consistent-type-assertions` with `assertionStyle: 'never'`)
   - Alphabetized imports, JSX props, object types, and object keys (perfectionist plugin)
   - React hooks rules (exhaustive-deps, rules-of-hooks)
   - Max 100 lines per file excluding blanks and comments (max-lines rule with `skipBlankLines: true, skipComments: true`)
   - No default exports (no-restricted-syntax targeting ExportDefaultDeclaration)

```javascript
import eslint from '@eslint/js';
import perfectionist from 'eslint-plugin-perfectionist';
import reactHooks from 'eslint-plugin-react-hooks';
import tseslint from 'typescript-eslint';

export default tseslint.config(
  { ignores: ['dist/**', 'node_modules/**'] },
  eslint.configs.recommended,
  ...tseslint.configs.strictTypeChecked,
  {
    languageOptions: {
      parserOptions: {
        projectService: true,
        tsconfigRootDir: import.meta.dirname,
      },
    },
    plugins: {
      perfectionist,
      'react-hooks': reactHooks,
    },
    rules: {
      '@typescript-eslint/consistent-type-assertions': [
        'error',
        { assertionStyle: 'never' },
      ],
      '@typescript-eslint/no-confusing-void-expression': 'off',
      '@typescript-eslint/no-explicit-any': 'error',
      '@typescript-eslint/no-unsafe-assignment': 'error',
      '@typescript-eslint/no-unsafe-call': 'error',
      '@typescript-eslint/no-unsafe-member-access': 'error',
      '@typescript-eslint/no-unsafe-return': 'error',
      '@typescript-eslint/restrict-template-expressions': [
        'error',
        { allowNumber: true, allowBoolean: true },
      ],
      'max-lines': [
        'error',
        { max: 100, skipBlankLines: true, skipComments: true },
      ],
      'no-restricted-syntax': [
        'error',
        {
          selector: 'ExportDefaultDeclaration',
          message: 'Default exports are not allowed. Use named exports.',
        },
      ],
      'perfectionist/sort-imports': 'error',
      'perfectionist/sort-jsx-props': 'error',
      'perfectionist/sort-object-types': 'error',
      'perfectionist/sort-objects': 'error',
      'react-hooks/exhaustive-deps': 'error',
      'react-hooks/rules-of-hooks': 'error',
    },
  },
);
```

Note: The `eslint.config.js` file itself is excluded from the `max-lines` rule because it is not in `src/`. ESLint is configured to lint `src/` only. If the config exceeds 100 lines, split into `eslint-rules.js` imported by `eslint.config.js`.
