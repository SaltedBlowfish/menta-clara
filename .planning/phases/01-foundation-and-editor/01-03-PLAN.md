2. Create `src/editor/use-editor-config.ts` -- custom hook that wraps TipTap's useEditor:
```typescript
import type { JSONContent } from '@tiptap/react';
import { useEditor } from '@tiptap/react';

import { editorExtensions } from './extensions';

interface UseEditorConfigOptions {
  content: JSONContent | null;
  onUpdate: (content: JSONContent) => void;
}

export function useEditorConfig(options: UseEditorConfigOptions) {
  const { content, onUpdate } = options;

  return useEditor({
    autofocus: true,
    content: content ?? undefined,
    extensions: editorExtensions,
    onUpdate: ({ editor }) => {
      onUpdate(editor.getJSON());
    },
  }, [content]);
}
````

Key points:

- `autofocus: true` per CONTEXT.md (editor auto-focuses on page load)
- Content is TipTap JSONContent (not markdown) per CONTEXT.md decision
- `onUpdate` callback passes `editor.getJSON()` (JSONContent) to parent
- The `[content]` dependency array ensures editor reinitializes when stored content loads
- No placeholder text per CONTEXT.md decision

3. Create `src/editor/editor.tsx` -- the EditorContent wrapper:

   ```typescript
   import type { JSONContent } from '@tiptap/react';
   import { EditorContent } from '@tiptap/react';

   import { useEditorConfig } from './use-editor-config';

   interface NoteEditorProps {
     content: JSONContent | null;
     onUpdate: (content: JSONContent) => void;
   }

   export function NoteEditor(props: NoteEditorProps) {
     const { content, onUpdate } = props;
     const editor = useEditorConfig({ content, onUpdate });

     if (!editor) {
       return null;
     }

     return <EditorContent editor={editor} />;
   }
   ```

   Guards against null editor (TipTap returns null on first render per RESEARCH.md Pitfall 4).

```typescript
import './app.css';

import { NoteEditor } from '../editor/editor';
import { useNote } from '../storage/use-note';
import { StorageWarning } from './storage-warning';

export function App() {
  const { content, error, loading, saveContent } = useNote();

  if (loading) {
    return null;
  }

  return (
    <>
      <div className="editor-container">
        <NoteEditor content={content} onUpdate={saveContent} />
      </div>
      {error ? <StorageWarning message={error} /> : null}
    </>
  );
}
````

Key behavior:

- Shows nothing while loading (avoids flash of empty editor before content loads)
- Renders NoteEditor with content from IndexedDB and saveContent callback
